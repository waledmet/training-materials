## أوراق الغش

### مرجع سريع لهياكل البيانات

**القوائم**:
```python
# الإنشاء
my_list = []
my_list = [1, 2, 3]

# الإضافة
my_list.append(4)       # إضافة إلى النهاية
my_list.insert(0, 0)    # إدراج في موقع
my_list.extend([5, 6])  # إضافة عدة عناصر

# الإزالة
my_list.remove(3)       # إزالة بالقيمة
my_list.pop()           # إزالة الأخير
my_list.pop(0)          # إزالة في الفهرس
my_list.clear()         # إزالة الكل

# الوصول
first = my_list[0]      # أول عنصر
last = my_list[-1]      # آخر عنصر
slice = my_list[1:3]    # تقطيع

# الترتيب
my_list.sort()          # ترتيب في المكان
my_list.reverse()       # عكس في المكان

# التحقق
if 5 in my_list:
    print("Found")

# التكرار
for item in my_list:
    print(item)

# القوائم المضغوطة
squares = [x**2 for x in range(10)]
```

**المجموعات**:
```python
# الإنشاء
my_tuple = ()
my_tuple = (1, 2, 3)
single = (1,)           # لاحظ الفاصلة!

# الوصول (نفس القوائم)
first = my_tuple[0]
last = my_tuple[-1]

# فك التعبئة
x, y, z = my_tuple

# لا يمكن التعديل!
# my_tuple[0] = 5      # خطأ!

# يمكن استخدامها في حلقات for
for item in my_tuple:
    print(item)
```

**القواميس**:
```python
# الإنشاء
my_dict = {}
my_dict = {"key": "value"}
my_dict = dict(key="value")

# الإضافة/التحديث
my_dict["new_key"] = "new_value"
my_dict.update({"key2": "value2"})

# الوصول
value = my_dict["key"]           # يرفع KeyError إذا لم يوجد
value = my_dict.get("key")       # يرجع None إذا لم يوجد
value = my_dict.get("key", "default")  # يرجع الافتراضي إذا لم يوجد

# الإزالة
del my_dict["key"]
value = my_dict.pop("key")       # إزالة وإرجاع
my_dict.clear()                  # إزالة الكل

# التحقق
if "key" in my_dict:
    print("Found")

# التكرار
for key in my_dict:
    print(key, my_dict[key])

for key, value in my_dict.items():
    print(key, value)

for key in my_dict.keys():
    print(key)

for value in my_dict.values():
    print(value)

# القواميس المضغوطة
squares = {x: x**2 for x in range(5)}
```

**المجموعات الفريدة**:
```python
# الإنشاء
my_set = set()
my_set = {1, 2, 3}

# الإضافة
my_set.add(4)

# الإزالة
my_set.remove(3)        # يرفع KeyError إذا لم يوجد
my_set.discard(3)       # لا خطأ إذا لم يوجد

# عمليات المجموعة
set1 = {1, 2, 3}
set2 = {3, 4, 5}

union = set1 | set2              # {1, 2, 3, 4, 5}
intersection = set1 & set2       # {3}
difference = set1 - set2         # {1, 2}
symmetric_diff = set1 ^ set2     # {1, 2, 4, 5}

# التحقق
if 3 in my_set:
    print("Found")

# التكرار
for item in my_set:
    print(item)
```

### مرجع سريع للدوال

```python
# دالة أساسية
def my_function():
    print("Hello")

# دالة مع معاملات
def greet(name):
    print(f"Hello, {name}")

# دالة مع إرجاع
def add(a, b):
    return a + b

# معاملات افتراضية
def greet(name, greeting="Hello"):
    print(f"{greeting}, {name}")

# قيم إرجاع متعددة
def get_stats(numbers):
    return min(numbers), max(numbers), sum(numbers)

# عدد متغير من الوسائط
def add_all(*args):
    return sum(args)

# وسائط كلمات مفتاحية
def person_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

# دالة Lambda
square = lambda x: x**2
```

```python
# try-except أساسي
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")

# استثناءات متعددة
try:
    num = int(input("Enter number: "))
    result = 10 / num
except ValueError:
    print("Invalid number")
except ZeroDivisionError:
    print("Cannot divide by zero")

# التقاط جميع الاستثناءات
try:
    # كود محفوف بالمخاطر
except Exception as e:
    print(f"Error: {e}")

# كتلة Finally
try:
    # كود
except Exception:
    # معالجة الخطأ
finally:
    # يعمل دائماً

# رفع الاستثناءات
if age < 0:
    raise ValueError("Age cannot be negative")
```

## الأخطاء الشائعة والحلول

### القوائم

**خطأ**: تعديل القائمة أثناء التكرار
```python
# خطأ
numbers = [1, 2, 3, 4, 5]
for num in numbers:
    if num % 2 == 0:
        numbers.remove(num)

# صحيح
numbers = [1, 2, 3, 4, 5]
numbers = [num for num in numbers if num % 2 != 0]
```

**خطأ**: نسخ القائمة
```python
# خطأ (كلاهما يشير إلى نفس القائمة)
list1 = [1, 2, 3]
list2 = list1

# صحيح
list1 = [1, 2, 3]
list2 = list1.copy()
```

### القواميس

**خطأ**: KeyError
```python
# خطأ
value = my_dict["key"]  # يرفع خطأ إذا لم يوجد المفتاح

# صحيح
value = my_dict.get("key", "default")
```

**خطأ**: التكرار والتعديل
```python
# خطأ
for key in my_dict:
    del my_dict[key]

# صحيح
for key in list(my_dict.keys()):
    del my_dict[key]
```

### الدوال

**خطأ**: وسائط افتراضية قابلة للتغيير
```python
# خطأ
def add_item(item, my_list=[]):
    my_list.append(item)
    return my_list

# صحيح
def add_item(item, my_list=None):
    if my_list is None:
        my_list = []
    my_list.append(item)
    return my_list
```
